   ```python
   class TeacherCNN(nn.Module):
       """Larger SimpleCNN variant: channels 64->128->256, classifier->512->10."""
       def __init__(self):
           super().__init__()
           self.features = nn.Sequential(
               nn.Conv2d(3, 64, kernel_size=3, padding=1),
               nn.ReLU(inplace=True),
               nn.MaxPool2d(2),
               nn.Conv2d(64, 128, kernel_size=3, padding=1),
               nn.ReLU(inplace=True),
               nn.MaxPool2d(2),
               nn.Conv2d(128, 256, kernel_size=3, padding=1),
               nn.ReLU(inplace=True),
               nn.MaxPool2d(2),
           )
           flat_dim = 256 * 4 * 4  # 4096 after 3x MaxPool2d on 32x32
           self.classifier = nn.Sequential(
               nn.Linear(flat_dim, 512),
               nn.ReLU(inplace=True),
               nn.Linear(512, NUM_CLASSES),
           )

       def forward(self, x):
           x = self.features(x)
           x = x.view(x.size(0), -1)
           return self.classifier(x)
   ```

   `StudentCNN(nn.Module)` -- Same architecture family but narrower:
   ```python
   class StudentCNN(nn.Module):
       """Smaller SimpleCNN variant: channels 16->32->64, classifier->128->10."""
       def __init__(self):
           super().__init__()
           self.features = nn.Sequential(
               nn.Conv2d(3, 16, kernel_size=3, padding=1),
               nn.ReLU(inplace=True),
               nn.MaxPool2d(2),
               nn.Conv2d(16, 32, kernel_size=3, padding=1),
               nn.ReLU(inplace=True),
   ```

   - `train_model(model, device, epochs, lr)` that trains with synthetic data from `get_sample_batch(batch_size=BATCH_SIZE, device=device)`, using `nn.CrossEntropyLoss()` and `torch.optim.Adam(model.parameters(), lr=lr)`. Logs progress every epoch. Uses a fixed set of synthetic batches (generate 5 batches, cycle through them each epoch).

   - `distill_knowledge(teacher, student, device, epochs, lr, temperature, alpha)` (per D-06):
     ```python
     def distill_knowledge(teacher, student, device, epochs, lr, temperature, alpha):
         teacher.eval()
         student.train()
         optimizer = torch.optim.Adam(student.parameters(), lr=lr)

         for epoch in range(epochs):
             inputs, labels = get_sample_batch(batch_size=BATCH_SIZE, device=device)
             student_logits = student(inputs)

             with torch.no_grad():
                 teacher_logits = teacher(inputs)

             # Soft target loss: KL divergence with temperature scaling (per D-06)
             soft_loss = F.kl_div(
                 F.log_softmax(student_logits / temperature, dim=1),
                 F.softmax(teacher_logits / temperature, dim=1),
                 reduction='batchmean'
             ) * (temperature * temperature)

             # Hard target loss: standard cross-entropy (per D-06)
             hard_loss = F.cross_entropy(student_logits, labels)

             # Combined loss (per D-06)
             loss = alpha * soft_loss + (1 - alpha) * hard_loss

             optimizer.zero_grad()
             loss.backward()
             optimizer.step()

             if (epoch + 1) % 2 == 0 or epoch == 0:
                 logger.info(f"  Distill epoch {epoch+1}/{epochs} | "
                           f"Loss: {loss.item():.4f} (soft: {soft_loss.item():.4f}, "
                           f"hard: {hard_loss.item():.4f})")
     ```
